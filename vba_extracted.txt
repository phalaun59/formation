Sub convert_CsvXls(ByVal chemin As String, ByVal fichier As String)

    Dim TB
    Dim Lig As Long, k As Integer
        
    Application.DisplayAlerts = False
    Application.ScreenUpdating = False
    
    NouvNom = Left(fichier, Len(fichier) - 3) & "xls"
    'Application.DisplayAlerts = False
    ActiveWorkbook.SaveAs chemin & NouvNom
    Workbooks(NouvNom).Close savechanges:=False
    Application.DisplayAlerts = True
    
    Set WbLu = Workbooks.Open(chemin & NouvNom)
       
    With ActiveSheet
    
        For Lig = 1 To Range("A65536").End(xlUp).Row
        'Changer la  , (virgule) par le séparateur de votre fichier
            TB = Split(.Cells(Lig, 1), ";")
            For k = 0 To UBound(TB)
                .Cells(Lig, k + 1) = Trim(TB(k))
            Next k
        Next Lig
    End With
    
    ActiveWorkbook.Save
    Application.DisplayAlerts = True
    Application.ScreenUpdating = True

End Sub

Function vider_repertoire(ByVal fichierCible As String, ByVal repertoire As String)
  Dim fichier As String
  Dim fs, f As Variant
  fichier = Dir(fichierCible)
  Do While fichier <> ""
    Set fs = CreateObject("Scripting.FileSystemObject")
    Set f = fs.GetFile(repertoire & fichier)
    f.Delete
    fichier = Dir
  Loop
End Function

Function timerProcess(ByVal pauseTime As Integer)
    Dim Start, Finish, TotalTime

    pauseTime = 5    ' Définit la durée.
    Start = Timer    ' Définit l'heure de début.
    Do While Timer < Start + pauseTime
        DoEvents    ' Donne le contrôle à d'autres processus.
    Loop
    Finish = Timer    ' Définit l'heure de fin.
    TotalTime = Finish - Start    ' Calcule la durée totale.
    
End Function

Function LireEcrireSql(ByVal RepBat As String, ByVal NomFichierLog As String, ByVal repRes As String, ByVal NomFichierReq As String, ByVal nomFichierSql As String)
    Dim fichier, repTmp, def As String
    Dim ExisteFichier As Boolean
    
    Dim flag As Integer
    
    
    flag = 0
    

    'If Not ExisteFichier Then
     ChDir ThisWorkbook.Path & "\REQUETES\"
     ' VERIFIER S IL Y A UNE REQUETE DE MODIFICATION/SUPPRESSION OU AJOUT
     NumFichier = FreeFile
    ' Open "C:\MOSAIC\MULTIREQUETE\REQUETES\" & NomFichierReq For Input As #NumFichier
    repTmp = ThisWorkbook.Path & "\TMP\"
     Open ThisWorkbook.Path & "\REQUETES\" & NomFichierReq For Input As #NumFichier
     Do While Not EOF(NumFichier)
        Line Input #NumFichier, chaine
        If (LCase(chaine) Like "*update*" Or LCase(chaine) Like "*insert*" Or LCase(chaine) Like "*delete*" Or LCase(chaine) Like "*merge*") Then
         flag = 1
            If commit = False Then
            MsgBox ("Vous n'etes pas autorisé a executer des requetes de modification,suppression ou ajout")
            End
         End If
        End If
     Loop
     Close #NumFichier
     
     ' CREATION DU SCRIPT SQL
     Set oFSR = CreateObject("Scripting.FileSystemObject")
     fichier = ThisWorkbook.Path & RepBat & "SCRIPT\" & nomFichierSql
     ExisteFichier = oFSR.FileExists(fichier)
    If Not ExisteFichier Then
        Set MonFichierReq = oFSR.CreateTextFile(fichier, ForWriting)
        MonFichierReq.WriteLine ("spool " & repRes & NomFichierLog)
        MonFichierReq.WriteLine ("whenever oserror exit rollback;")
        MonFichierReq.WriteLine ("whenever sqlerror exit rollback;")
        MonFichierReq.WriteLine ("SET pages 10000")
        MonFichierReq.WriteLine ("SET LINES 500")
        MonFichierReq.WriteLine ("SET LINESIZE 32767")
        ' MonFichierReq.WriteLine ("SET pagesize 5000")
        MonFichierReq.WriteLine ("SET serveroutput ON size 1000000")
        MonFichierReq.WriteLine ("SET verify off")
        MonFichierReq.WriteLine ("set TRIMSPOOL ON")
        ' MonFichierReq.WriteLine ("set EMBEDDED OFF")
        ' MonFichierReq.WriteLine ("set NEWP NONE")
        ' MonFichierReq.WriteLine ("set PAGESIZE 0")
        If flag = 0 Then
            MonFichierReq.WriteLine ("set FEEDBACK OFF")
            Else: MonFichierReq.WriteLine ("set FEEDBACK ON")
            End If
       
        def = "SET SQLFORMAT DELIMITED ; " & Chr(34) & Chr(32) & Chr(34)
        MonFichierReq.WriteLine (def)
        ' MonFichierReq.WriteLine ("set TAB OFF")
        NumFichier = FreeFile
        
       ' Open "C:\MOSAIC\MULTIREQUETE\REQUETES\" & NomFichierReq For Input As #NumFichier
        Open ThisWorkbook.Path & "\REQUETES\" & NomFichierReq For Input As #NumFichier
        Do While Not EOF(NumFichier)
           Line Input #NumFichier, chaine
           MonFichierReq.WriteLine (chaine)
        Loop
        Close #NumFichier
        
        
        MonFichierReq.WriteLine ("spool off")
        
        If flag = 1 Then
        MonFichierReq.WriteLine ("commit;")
        End If
        'MonFichier.WriteLine ("commit;")
        
        MonFichierReq.WriteLine ("spool " & repTmp & NomFichierLog & ".terminer")
        MonFichierReq.WriteLine ("whenever oserror exit rollback;")
        MonFichierReq.WriteLine ("whenever sqlerror exit rollback;")
        MonFichierReq.WriteLine ("set heading off")
        
        MonFichierReq.WriteLine ("spool off")
        MonFichierReq.WriteLine ("exit")
        MonFichierReq.Close
     End If
End Function

Private Sub BTASK_Click()
    Dim Modru, modrp, MODALBIS, fichierSchedule, Modalite, execbat As String
    Dim incomplet As Boolean
    repSchedule = ThisWorkbook.Path & "\BAT_BOT\"
    fichierSchedule = repSchedule & "schedule.bat"
    Modru = " /ru system "
    result = vider_repertoire(repSchedule & "schedule.bat", repSchedule)
    execbat = "nodepad.exe"
    MODALBIS = ""
    Modalite = ""
    incomplet = False
    
    If TTask.Text = "" Then
       MsgBox ("le nom de la tache doit etre renseigné")
        incomplet = True
        
    Else
        If CPLANIF.Value = "" Then
             MsgBox ("la frequence de planification doit etre renseigné")
            incomplet = True
        Else
            If Tlogin.Text = "" Then
                MsgBox ("Vous n'avez pas renseigné le login pour le lancement de la tache")
                incomplet = True
            End If
        End If
    
    End If
    If incomplet = False Then execbat = lancer_script(False, TTask.Text)
    If execbat = "" Then
        incomplet = True
    Else
        repSchedule = ThisWorkbook.Path & "\BAT_BOT\"
        fichierSchedule = repSchedule & "schedule.bat"
        result = vider_repertoire(repSchedule & "schedule.bat", repSchedule)
        Set oFS = CreateObject("Scripting.FileSystemObject")
        Set MonFichierTask = oFS.CreateTextFile(fichierSchedule, ForWriting)

        MonFichierTask.WriteLine (cmd)
        MonFichierTask.Close
        retval = Shell(fichierSchedule, vbHide)
    End If
    
   
    If incomplet = False Then
        If StrComp(CPLANIF.Value, "MINUTE", vbBinaryCompare) = 0 Then
    
        Else
            If StrComp(CPLANIF.Value, "HOURLY", vbBinaryCompare) = 0 Then
    
            Else
                If StrComp(CPLANIF.Value, "DAILY", vbBinaryCompare) = 0 Then
    
                 Else
                    If StrComp(CPLANIF.Value, "TOUTES LES SEMAINES", vbBinaryCompare) = 0 Then
                        For i = 0 To 6
                            If TAB_JOUR(i) <> "" Then
                                If MODALBIS = "" Then
                                    MODALBIS = MODALBIS & TAB_JOUR(i)
                                Else
                                    MODALBIS = MODALBIS & "," & TAB_JOUR(i)
                                End If
                            End If
                        Next i
                        If MODALBIS <> "" Then
                                 MODALBIS = "/D " & MODALBIS
                         Else
                            MsgBox ("au moins un jours doit etre selectionné")
                            End
                         End If
                     Else
                        If StrComp(CPLANIF.Value, "MENSUEL", vbBinaryCompare) = 0 Then
                            
                                For i = 0 To 11
                                    If TAB_MOIS(i) <> "" Then
                                       If MODALBIS = "" Then
                                           MODALBIS = MODALBIS & TAB_MOIS(i)
                                       Else
                                           MODALBIS = MODALBIS & "," & TAB_MOIS(i)
                                       End If
                                    End If
                                Next i
                                If MODALBIS <> "" Then
                                    MODALBIS = "/M " & MODALBIS
                                Else
                                    MsgBox ("au moins un mois doit etre selectionné")
                                    End
                                End If
                             
                        Else
    
                        End If
                     End If
                 End If
            End If
        End If
        If Modalite = "" Then
         Modalite = ""
        Else
            Modalite = " /MO " & Modalite
        End If
        If Tlogin.Text <> "" Then
           Modru = " /ru " & Tlogin.Text
           modrp = " /rp " & Tpassword.Text
        End If
        result = vider_repertoire(repSchedule & "schedule.bat", repSchedule)
        
        Set oFS = CreateObject("Scripting.FileSystemObject")
        Set MonFichierBat = oFS.CreateTextFile(fichierSchedule, ForWriting)

Private Sub CPLANIF_Change()
    CModif.Clear
    
    If StrComp(CPLANIF.Value, "MINUTE", vbBinaryCompare) = 0 Then
         LModif.Visible = True
         CModif.Visible = True
    Else
        If StrComp(CPLANIF.Value, "HOURLY", vbBinaryCompare) = 0 Then
            LModif.Visible = True
            CModif.Visible = True
            
            For i = 1 To 23
               CModif.AddItem i
            Next i
            CModif.Value = 1
        Else
            If StrComp(CPLANIF.Value, "DAILY", vbBinaryCompare) = 0 Then
                For i = 1 To 365
                   CModif.AddItem i
                Next i
                LModif.Visible = True
                CModif.Visible = True
             Else
                If StrComp(CPLANIF.Value, "TOUTES LES SEMAINES", vbBinaryCompare) = 0 Then
                    For i = 1 To 52
                        CModif.AddItem i
                    Next i
                    FJOUR.Top = FMOIS.Top
                    FJOUR.Left = FMOIS.Left
                    FJOUR.Visible = True
                    FMOIS.Visible = False
                    
                    LModif.Visible = True
                    CModif.Visible = True
                 Else
                    If StrComp(CPLANIF.Value, "MENSUEL", vbBinaryCompare) = 0 Then
'                        For i = 1 To 12
'                            CModif.AddItem
'                        Next i
                        CModif.AddItem "PREMIER"
                        CModif.AddItem "SECOND"
                        CModif.AddItem "TROISIEME"
                        CModif.AddItem "QUATRIEME"
                        CModif.AddItem "DERNIER"
                        
                        LModif.Visible = True
                        CModif.Visible = True
                        FJOUR.Visible = False
                        FMOIS.Visible = True
                    Else
                        LModif.Visible = False
                        CModif.Visible = False
                    End If
                 End If
             End If
        End If
    End If
            
               
                    

    
End Sub
